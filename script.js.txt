// Close the popup
function closeRecipe() {
  document.getElementById("recipe").style.display = "none";
}

// Play or pause background music
function toggleMusic() {
  const music = document.getElementById("bgMusic");
  if (music.paused) {
    music.play();
  } else {
    music.pause();
  }
}
